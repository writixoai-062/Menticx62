```javascript
function toggleSbSection(id, iconId) {
  const el = document.getElementById(id);
  const icon = document.getElementById(iconId);
  if (!el) return;
  const isHidden = el.style.display === 'none' || el.style.display === '';
  el.style.display = isHidden ? 'block' : 'none';
```
